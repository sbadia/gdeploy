      def key_length(*v)
        @key_length = nil if !defined?(@key_length)
        if v.empty?
          @key_length = superclass.key_length if @key_length.nil? && superclass.respond_to?(:key_length)
          return @key_length
        elsif v.length == 1
          @key_length = v.first
        else
          raise ArgumentError, "wrong number of arguments (#{v.length} for 1)"
        end
      end
